### 18.2 评论交互逻辑

```javascript
// pages/detail/detail.js —— 追加评论相关方法

data: {
  // ... 之前的数据
  commentText: '',
  replyTarget: null,     // 回复目标 { id, nickname }
  showCommentInput: false
},

// 加载评论列表
async loadComments(articleId) {
  try {
    const comments = await get(`/comments/${articleId}`)
    this.setData({ comments })
  } catch (e) {
    console.error('加载评论失败', e)
  }
},

// 评论输入
onCommentInput(e) {
  this.setData({ commentText: e.detail.value })
},

// 点击回复
replyComment(e) {
  const comment = e.currentTarget.dataset.comment
  this.setData({
    replyTarget: { id: comment.id, nickname: comment.nickname },
    showCommentInput: true
  })
},

// 提交评论
async submitComment() {
  if (!this.checkLogin()) return
  const content = this.data.commentText.trim()
  if (!content) {
    wx.showToast({ title: '请输入评论内容', icon: 'none' })
    return
  }

  try {
    const data = {
      articleId: this.data.article.id,
      content: content
    }
    if (this.data.replyTarget) {
      data.parentId = this.data.replyTarget.id
    }

    await post('/comments', data)

    wx.showToast({ title: '评论成功', icon: 'success' })
    this.setData({
      commentText: '',
      replyTarget: null,
      showCommentInput: false
    })

    // 重新加载评论列表
    this.loadComments(this.data.article.id)
  } catch (e) {
    wx.showToast({ title: '评论失败', icon: 'none' })
  }
},

// 点赞评论
async likeComment(e) {
  if (!this.checkLogin()) return
  const commentId = e.currentTarget.dataset.id

  try {
    await post(`/comments/like/${commentId}`)
    // 更新本地评论状态
    const comments = this.data.comments.map(c => {
      if (c.id === commentId) {
        return {
          ...c,
          isLiked: !c.isLiked,
          likeCount: c.isLiked ? c.likeCount - 1 : c.likeCount + 1
        }
      }
      return c
    })
    this.setData({ comments })
  } catch (e) {
    console.error('点赞失败', e)
  }
}
```

> **Java开发者注意点**：嵌套评论的实现方式与 Java 后端密切相关。常见的方案有两种：
>
> 1. **邻接表模式**（本教程采用）：每条评论有 `parent_id` 字段指向父评论，后端通过递归或两次查询组装树形结构。这类似于 Java 中的 `Comment` 实体包含 `List<Comment> children` 字段。
>
> 2. **路径枚举模式**：使用 `path` 字段存储如 `1/5/12` 的路径，通过 `LIKE '1/5/%'` 查询所有子孙评论。
>
> 在 Spring Boot 中，评论树组装通常在 Service 层完成：
> ```java
> private List<CommentVO> buildCommentTree(List<Comment> comments) {
>     Map<Long, List<Comment>> childrenMap = comments.stream()
>         .filter(c -> c.getParentId() != null)
>         .collect(Collectors.groupingBy(Comment::getParentId));
>     return comments.stream()
>         .filter(c -> c.getParentId() == null)
>         .map(c -> convertToTree(c, childrenMap))
>         .collect(Collectors.toList());
> }
> ```
