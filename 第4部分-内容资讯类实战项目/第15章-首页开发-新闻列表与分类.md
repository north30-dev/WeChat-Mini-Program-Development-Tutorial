## 第15章 首页开发：新闻列表与分类

首页是用户打开小程序后看到的第一个页面，需要实现分类导航、新闻列表、下拉刷新和上拉加载。这些功能在 Java Web 开发中通常由一个带有分页和筛选功能的 `NewsController` 配合 Thymeleaf 模板完成，而小程序中则完全由前端驱动。

### 15.2 请求工具封装

```javascript
// utils/request.js
const BASE_URL = 'https://your-domain.com/api'

const request = (options) => {
  return new Promise((resolve, reject) => {
    const token = wx.getStorageSync('token')
    wx.request({
      url: BASE_URL + options.url,
      method: options.method || 'GET',
      data: options.data || {},
      header: {
        'Content-Type': 'application/json',
        'Authorization': token ? `Bearer ${token}` : ''
      },
      success: (res) => {
        if (res.statusCode === 200) {
          if (res.data.code === 0) {
            resolve(res.data.data)
          } else {
            wx.showToast({ title: res.data.message, icon: 'none' })
            reject(res.data)
          }
        } else if (res.statusCode === 401) {
          // token过期，跳转登录
          wx.removeStorageSync('token')
          wx.navigateTo({ url: '/pages/profile/profile' })
          reject(new Error('未授权'))
        } else {
          wx.showToast({ title: '网络异常', icon: 'none' })
          reject(new Error('网络异常'))
        }
      },
      fail: (err) => {
        wx.showToast({ title: '网络连接失败', icon: 'none' })
        reject(err)
      }
    })
  })
}

// 导出便捷方法
const get = (url, data) => request({ url, method: 'GET', data })
const post = (url, data) => request({ url, method: 'POST', data })

module.exports = { request, get, post }
```

> **类比理解**：这个 `request.js` 就像你在 Java Web 前端项目中封装的 Axios 实例，添加了请求拦截器（自动携带 token）和响应拦截器（统一错误处理）。`wx.getStorageSync('token')` 类似于从 `localStorage` 中读取 token。

```javascript
// pages/index/index.js
const { get } = require('../../utils/request')

Page({
  data: {
    categories: [],
    currentCategory: 0,    // 0表示"全部"
    articles: [],
    page: 1,
    pageSize: 10,
    loading: false,
    noMore: false,
    showSkeleton: true
  },

  onLoad() {
    this.loadCategories()
    this.loadArticles()
  },

  // 下拉刷新
  onPullDownRefresh() {
    this.setData({ page: 1, articles: [], noMore: false })
    this.loadArticles().then(() => {
      wx.stopPullDownRefresh()
    })
  },

  // 上拉加载更多（触底）
  onReachBottom() {
    if (this.data.loading || this.data.noMore) return
    this.loadArticles()
  },

  // 加载分类
  async loadCategories() {
    try {
      const categories = await get('/categories')
      this.setData({ categories: [{ id: 0, name: '全部' }, ...categories] })
    } catch (e) {
      console.error('加载分类失败', e)
    }
  },

  // 加载文章列表
  async loadArticles() {
    if (this.data.loading) return
    this.setData({ loading: true })

    try {
      const params = {
        page: this.data.page,
        pageSize: this.data.pageSize
      }
      if (this.data.currentCategory > 0) {
        params.categoryId = this.data.currentCategory
      }

      const result = await get('/articles', params)

      const newArticles = this.data.page === 1
        ? result.records
        : this.data.articles.concat(result.records)

      this.setData({
        articles: newArticles,
        showSkeleton: false,
        loading: false,
        noMore: newArticles.length >= result.total
      })

      if (!this.data.noMore) {
        this.setData({ page: this.data.page + 1 })
      }
    } catch (e) {
      this.setData({ loading: false, showSkeleton: false })
      console.error('加载文章失败', e)
    }
  },

  // 切换分类
  switchCategory(e) {
    const id = e.currentTarget.dataset.id
    if (id === this.data.currentCategory) return
    this.setData({
      currentCategory: id,
      page: 1,
      articles: [],
      noMore: false,
      showSkeleton: true
    })
    this.loadArticles()
  },

  // 跳转详情
  goDetail(e) {
    const id = e.currentTarget.dataset.id
    wx.navigateTo({ url: `/pages/detail/detail?id=${id}` })
  },

  // 跳转搜索
  goSearch() {
    wx.navigateTo({ url: '/pages/search/search' })
  }
})
```

> **Java开发者注意点**：分页参数 `page` 和 `pageSize` 与 Java 后端的 `PageHelper` 或 MyBatis-Plus 的 `Page<T>` 对象完全对应。后端返回的分页结构通常为 `{ records: [], total: 100, page: 1, pageSize: 10 }`，前端通过 `concat` 方法追加数据实现"加载更多"——这和你在 Web 前端使用无限滚动（Infinite Scroll）的原理一致。

> **类比理解**：`onPullDownRefresh` 相当于用户在浏览器中按 F5 刷新页面；`onReachBottom` 相当于滚动到页面底部触发 AJAX 请求加载下一页。`enablePullDownRefresh: true` 类似于在 HTML 中注册 `window.addEventListener('scroll', ...)` 事件。
