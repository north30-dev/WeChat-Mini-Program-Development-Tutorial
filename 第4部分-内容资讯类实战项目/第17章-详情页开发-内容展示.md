## 第17章 详情页开发：内容展示

详情页是新闻类应用的核心页面，需要处理富文本渲染、图片预览、字体调节和夜间模式等功能。

### 17.1 页面参数接收

```javascript
// pages/detail/detail.js
const { get, post } = require('../../utils/request')

Page({
  data: {
    article: null,
    comments: [],
    fontSize: 16,       // 字体大小（px）
    isNight: false,      // 夜间模式
    isFavorited: false,  // 是否已收藏
    contentNodes: []     // 富文本节点
  },

  onLoad(options) {
    // 接收页面参数
    const id = options.id
    if (id) {
      this.loadArticle(id)
      this.loadComments(id)
      this.checkFavorite(id)
    }

    // 读取用户设置的字体大小
    const fontSize = wx.getStorageSync('fontSize') || 16
    this.setData({ fontSize })

    // 读取夜间模式设置
    const isNight = wx.getStorageSync('nightMode') || false
    this.setData({ isNight })
  },

  async loadArticle(id) {
    try {
      const article = await get(`/articles/${id}`)
      this.setData({ article })

      // 设置页面标题
      wx.setNavigationBarTitle({ title: article.title })

      // 记录阅读历史
      this.saveHistory(article)
    } catch (e) {
      wx.showToast({ title: '加载失败', icon: 'none' })
    }
  },

  // ... 其他方法后续补充
})
```

> **Java开发者注意点**：`onLoad(options)` 中的 `options.id` 类似于 Spring MVC 中 `@PathVariable` 或 `@RequestParam` 接收的路径参数。小程序通过页面 URL 传参：`/pages/detail/detail?id=123`，与 Java Web 中 `/articles/123` 的路由参数机制一致。`wx.setNavigationBarTitle` 则相当于动态设置页面的 `<title>` 标签。

### 17.3 图片预览与字体调节

```javascript
// pages/detail/detail.js —— 追加方法

// 点击富文本内容中的图片，触发预览
onContentTap(e) {
  // rich-text组件的bindtap事件无法直接获取点击的图片
  // 推荐使用mp-html组件替代（见下方说明）
},

// 切换夜间模式
toggleNight() {
  const isNight = !this.data.isNight
  this.setData({ isNight })
  wx.setStorageSync('nightMode', isNight)

  // 同步修改导航栏颜色
  wx.setNavigationBarColor({
    frontColor: isNight ? '#ffffff' : '#000000',
    backgroundColor: isNight ? '#1a1a1a' : '#ffffff'
  })
},

// 字体大小调节（弹出选择器）
adjustFont() {
  const fontSizeList = [14, 16, 18, 20, 22]
  const currentIndex = fontSizeList.indexOf(this.data.fontSize)

  wx.showActionSheet({
    itemList: fontSizeList.map(s => `${s}px`),
    success: (res) => {
      const newFontSize = fontSizeList[res.tapIndex]
      this.setData({ fontSize: newFontSize })
      wx.setStorageSync('fontSize', newFontSize)
    }
  })
},

// 收藏切换
async toggleFavorite() {
  if (!this.checkLogin()) return

  try {
    if (this.data.isFavorited) {
      await post(`/favorites/remove`, { articleId: this.data.article.id })
      this.setData({ isFavorited: false })
      wx.showToast({ title: '已取消收藏', icon: 'none' })
    } else {
      await post(`/favorites/add`, { articleId: this.data.article.id })
      this.setData({ isFavorited: true })
      wx.showToast({ title: '已收藏', icon: 'success' })
    }
  } catch (e) {
    console.error('收藏操作失败', e)
  }
},

// 检查登录状态
checkLogin() {
  const token = wx.getStorageSync('token')
  if (!token) {
    wx.showModal({
      title: '提示',
      content: '请先登录',
      confirmText: '去登录',
      success: (res) => {
        if (res.confirm) {
          wx.switchTab({ url: '/pages/profile/profile' })
        }
      }
    })
    return false
  }
  return true
},

// 检查收藏状态
async checkFavorite(articleId) {
  const token = wx.getStorageSync('token')
  if (!token) return
  try {
    const isFavorited = await get(`/favorites/check/${articleId}`)
    this.setData({ isFavorited })
  } catch (e) {
    // 未登录或接口异常，忽略
  }
},

// 保存阅读历史
saveHistory(article) {
  let history = wx.getStorageSync('readHistory') || []
  history = history.filter(item => item.id !== article.id)
  history.unshift({
    id: article.id,
    title: article.title,
    coverUrl: article.coverUrl,
    readTime: new Date().toISOString()
  })
  if (history.length > 50) history = history.slice(0, 50)
  wx.setStorageSync('readHistory', history)
},

// 分享给好友
onShareAppMessage() {
  return {
    title: this.data.article.title,
    path: `/pages/detail/detail?id=${this.data.article.id}`,
    imageUrl: this.data.article.coverUrl
  }
},

// 分享到朋友圈
onShareTimeline() {
  return {
    title: this.data.article.title,
    query: `id=${this.data.article.id}`
  }
}
```

> **Java开发者注意点**：`rich-text` 组件虽然能渲染 HTML，但存在以下限制：不支持点击事件冒泡、不支持懒加载、样式兼容性有限。**推荐使用开源组件 `mp-html`**（通过 npm 安装），它提供了更完善的富文本支持，包括图片预览、代码高亮、表格渲染等。安装方式：
>
> ```
> npm install mp-html
> ```
>
> 在页面 JSON 中引入：
> ```json
> {
>   "usingComponents": {
>     "mp-html": "mp-html/index"
>   }
> }
> ```
>
> WXML 中使用：
> ```xml
> <mp-html content="{{article.content}}" bindimgtap="onImageTap" />
> ```
>
> JS 中处理图片预览：
> ```javascript
> onImageTap(e) {
>   wx.previewImage({
>     current: e.detail.src,
>     urls: this.data.article.images  // 文章中所有图片URL数组
>   })
> }
> ```
>
> 这类似于 Java Web 中使用 `@GetMapping("/images/{id}")` 返回图片流，前端通过 `<img>` 标签展示，再用 Lightbox.js 实现图片预览。
