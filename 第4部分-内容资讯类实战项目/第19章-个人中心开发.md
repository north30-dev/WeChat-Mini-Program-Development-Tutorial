```javascript
// pages/profile/profile.js
const { get, post } = require('../../utils/request')

const app = getApp()

Page({
  data: {
    hasLogin: false,
    userInfo: {},
    favoriteCount: 0,
    historyCount: 0,
    isNightMode: false,
    fontSize: 16
  },

  onShow() {
    // 每次显示页面时刷新状态（从其他页面返回时）
    this.checkLoginStatus()
    this.loadUserStats()
    this.loadSettings()
  },

  // 检查登录状态
  checkLoginStatus() {
    const token = wx.getStorageSync('token')
    const userInfo = wx.getStorageSync('userInfo')
    this.setData({
      hasLogin: !!token,
      userInfo: userInfo || {}
    })
  },

  // 加载用户统计
  loadUserStats() {
    const token = wx.getStorageSync('token')
    if (!token) return

    // 收藏数
    get('/favorites/count').then(count => {
      this.setData({ favoriteCount: count })
    }).catch(() => {})

    // 阅读历史数
    const history = wx.getStorageSync('readHistory') || []
    this.setData({ historyCount: history.length })
  },

  // 加载设置
  loadSettings() {
    this.setData({
      isNightMode: wx.getStorageSync('nightMode') || false,
      fontSize: wx.getStorageSync('fontSize') || 16
    })
  },

  // 微信登录
  wxLogin() {
    wx.login({
      success: (loginRes) => {
        if (loginRes.code) {
          this.loginToServer(loginRes.code)
        }
      },
      fail: () => {
        wx.showToast({ title: '登录失败', icon: 'none' })
      }
    })
  },

  // 向后端换取token
  async loginToServer(code) {
    wx.showLoading({ title: '登录中...' })
    try {
      const result = await post('/auth/login', { code })
      wx.setStorageSync('token', result.token)
      wx.setStorageSync('userInfo', result.userInfo)
      this.setData({
        hasLogin: true,
        userInfo: result.userInfo
      })
      wx.showToast({ title: '登录成功', icon: 'success' })
    } catch (e) {
      wx.showToast({ title: '登录失败', icon: 'none' })
    } finally {
      wx.hideLoading()
    }
  },

  // 退出登录
  logout() {
    wx.showModal({
      title: '提示',
      content: '确定要退出登录吗？',
      success: (res) => {
        if (res.confirm) {
          wx.removeStorageSync('token')
          wx.removeStorageSync('userInfo')
          this.setData({ hasLogin: false, userInfo: {}, favoriteCount: 0 })
          wx.showToast({ title: '已退出', icon: 'success' })
        }
      }
    })
  },

  // 跳转收藏列表
  goFavorites() {
    if (!this.checkLogin()) return
    wx.navigateTo({ url: '/pages/favorites/favorites' })
  },

  // 跳转阅读历史
  goHistory() {
    wx.navigateTo({ url: '/pages/history/history' })
  },

  // 夜间模式切换
  onNightModeChange(e) {
    const isNightMode = e.detail.value
    this.setData({ isNightMode })
    wx.setStorageSync('nightMode', isNightMode)
  },

  // 字体大小调节
  adjustFontSize() {
    const fontSizeList = [14, 16, 18, 20, 22]
    const currentIndex = fontSizeList.indexOf(this.data.fontSize)
    wx.showActionSheet({
      itemList: fontSizeList.map(s => `${s}px`),
      success: (res) => {
        const newFontSize = fontSizeList[res.tapIndex]
        this.setData({ fontSize: newFontSize })
        wx.setStorageSync('fontSize', newFontSize)
      }
    })
  },

  // 清除缓存
  clearCache() {
    wx.showModal({
      title: '提示',
      content: '确定要清除所有本地缓存吗？（不会影响登录状态）',
      success: (res) => {
        if (res.confirm) {
          wx.removeStorageSync('search_history')
          wx.removeStorageSync('readHistory')
          wx.removeStorageSync('fontSize')
          wx.removeStorageSync('nightMode')
          this.setData({ historyCount: 0, fontSize: 16, isNightMode: false })
          wx.showToast({ title: '缓存已清除', icon: 'success' })
        }
      }
    })
  },

  // 关于我们
  showAbout() {
    wx.showModal({
      title: '关于我们',
      content: '新闻资讯 v1.0.0\n一款简洁的新闻阅读小程序',
      showCancel: false
    })
  },

  checkLogin() {
    if (!this.data.hasLogin) {
      wx.showModal({
        title: '提示',
        content: '请先登录',
        confirmText: '去登录',
        success: (res) => {
          if (res.confirm) this.wxLogin()
        }
      })
      return false
    }
    return true
  }
})
```

```javascript
// pages/favorites/favorites.js
const { get, post } = require('../../utils/request')

Page({
  data: { list: [] },

  onShow() {
    this.loadFavorites()
  },

  async loadFavorites() {
    try {
      const list = await get('/favorites/list')
      this.setData({ list })
    } catch (e) {
      console.error('加载收藏失败', e)
    }
  },

  async removeFavorite(e) {
    const id = e.currentTarget.dataset.id
    try {
      await post('/favorites/remove', { id })
      wx.showToast({ title: '已取消收藏', icon: 'none' })
      this.loadFavorites()
    } catch (e) {
      wx.showToast({ title: '操作失败', icon: 'none' })
    }
  },

  goDetail(e) {
    const id = e.currentTarget.dataset.id
    wx.navigateTo({ url: `/pages/detail/detail?id=${id}` })
  }
})
```

> **Java开发者注意点**：小程序的页面生命周期中，`onShow` 会在每次页面显示时调用（包括从其他页面返回），而 `onLoad` 只在页面首次创建时调用一次。这类似于 Java Web 中 `@GetMapping` 的每次请求都会执行，而 `@PostConstruct` 只在 Bean 初始化时执行一次。因此，收藏列表放在 `onShow` 中加载，确保用户从详情页取消收藏返回后列表能自动刷新。
