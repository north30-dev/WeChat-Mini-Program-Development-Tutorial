```javascript
// pages/search/search.js
const { get } = require('../../utils/request')
const HISTORY_KEY = 'search_history'
const MAX_HISTORY = 10

Page({
  data: {
    keyword: '',
    autoFocus: true,
    searched: false,
    results: [],
    history: [],
    hotKeywords: ['人工智能', 'Spring Boot', '前端开发', '云计算', '大数据', '区块链'],
    timer: null
  },

  onLoad() {
    this.loadHistory()
  },

  onUnload() {
    // 页面卸载时清除防抖定时器
    if (this.data.timer) clearTimeout(this.data.timer)
  },

  // 输入事件（防抖处理）
  onInput(e) {
    const keyword = e.detail.value.trim()
    this.setData({ keyword })

    // 防抖：500ms内无新输入才触发搜索
    if (this.data.timer) clearTimeout(this.data.timer)
    if (keyword.length === 0) {
      this.setData({ searched: false, results: [] })
      return
    }
    this.data.timer = setTimeout(() => {
      this.doSearch()
    }, 500)
  },

  // 执行搜索
  async doSearch() {
    const keyword = this.data.keyword.trim()
    if (!keyword) return

    if (this.data.timer) clearTimeout(this.data.timer)

    this.setData({ searched: true, results: [] })

    try {
      const results = await get('/articles/search', { keyword, page: 1, pageSize: 20 })
      this.setData({ results })
      this.saveHistory(keyword)
    } catch (e) {
      console.error('搜索失败', e)
    }
  },

  // 通过标签搜索
  searchByTag(e) {
    const keyword = e.currentTarget.dataset.keyword
    this.setData({ keyword })
    this.doSearch()
  },

  // 清空关键词
  clearKeyword() {
    this.setData({ keyword: '', searched: false, results: [] })
  },

  // 加载搜索历史
  loadHistory() {
    const history = wx.getStorageSync(HISTORY_KEY) || []
    this.setData({ history })
  },

  // 保存搜索历史
  saveHistory(keyword) {
    let history = wx.getStorageSync(HISTORY_KEY) || []
    // 去重
    history = history.filter(item => item !== keyword)
    // 插入到最前面
    history.unshift(keyword)
    // 限制数量
    if (history.length > MAX_HISTORY) {
      history = history.slice(0, MAX_HISTORY)
    }
    wx.setStorageSync(HISTORY_KEY, history)
    this.setData({ history })
  },

  // 清除搜索历史
  clearHistory() {
    wx.removeStorageSync(HISTORY_KEY)
    this.setData({ history: [] })
  },

  // 跳转详情
  goDetail(e) {
    const id = e.currentTarget.dataset.id
    wx.navigateTo({ url: `/pages/detail/detail?id=${id}` })
  },

  // 返回
  goBack() {
    wx.navigateBack()
  }
})
```

> **Java开发者注意点**：搜索防抖（Debounce）的实现原理与 Java Web 前端完全一致——利用 `setTimeout` 延迟执行，每次新输入时清除上一个定时器。在 Java 中你可能用 `ScheduledExecutorService` 或 Guava 的 `RateLimiter` 实现类似效果。搜索历史使用 `wx.setStorageSync` 存储在本地，相当于浏览器的 `localStorage`。

### 16.2 首页轮播图

在首页 `index.wxml` 中添加轮播图组件：

```xml
<!-- pages/index/index.wxml —— 在搜索栏和分类Tab之间插入 -->
<!-- 轮播Banner -->
<swiper
  wx:if="{{banners.length > 0}}"
  class="banner-swiper"
  indicator-dots
  indicator-color="rgba(255,255,255,0.5)"
  indicator-active-color="#fff"
  autoplay
  interval="4000"
  circular
  bindchange="onBannerChange"
>
  <swiper-item wx:for="{{banners}}" wx:key="id">
    <view class="banner-item" bindtap="goDetail" data-id="{{item.id}}">
      <image class="banner-image" src="{{item.coverUrl}}" mode="aspectFill" />
      <view class="banner-title-wrap">
        <text class="banner-title">{{item.title}}</text>
      </view>
    </view>
  </swiper-item>
</swiper>
```

> **类比理解**：`swiper` 组件类似于 Web 前端的 Swiper.js 或 Bootstrap Carousel。`autoplay`、`interval`、`circular` 等属性与 JavaScript 轮播库的配置项一一对应。`indicator-dots` 控制底部指示点显示，`bindchange` 监听滑动事件。
