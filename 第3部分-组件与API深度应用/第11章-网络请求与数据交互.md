## 第11章 网络请求与数据交互

网络请求是小程序与 Java 后端通信的桥梁。`wx.request` 是唯一的网络请求 API，类似于浏览器中的 `XMLHttpRequest` 或 `fetch`，但使用方式和配置有所不同。

### 11.1 wx.request 基础用法

```javascript
// 最基础的请求方式
wx.request({
  url: 'https://api.example.com/articles',
  method: 'GET',
  data: { page: 1, size: 10 },
  header: {
    'content-type': 'application/json',
    'Authorization': 'Bearer eyJhbGciOiJIUzI1NiJ9...'
  },
  success(res) {
    console.log('响应数据：', res.data)
    console.log('状态码：', res.statusCode)
  },
  fail(err) {
    console.error('请求失败：', err)
  },
  complete() {
    console.log('请求完成（无论成功或失败）')
  }
})
```

> **Java开发者注意点**：`wx.request` 与你在 Java 后端使用的 `RestTemplate` 或 `OkHttpClient` 有几个关键差异：
> 1. **域名限制**：所有请求域名必须在小程序管理后台配置白名单，且必须是 HTTPS。开发阶段可以在开发者工具中勾选"不校验合法域名"。
> 2. **Cookie 机制**：`wx.request` 默认不发送和保存 Cookie。如果后端依赖 Cookie/Session，需要手动管理。
> 3. **并发限制**：同时最多 10 个 wx.request 连接（类似浏览器同域并发限制）。
> 4. **超时时间**：默认 60 秒，可通过 `timeout` 参数自定义。

### 11.2 Promise 化封装

回调地狱（Callback Hell）在复杂业务中难以维护。我们用 Promise 对 `wx.request` 进行封装，使代码风格更接近 `async/await`。

```javascript
// utils/request.js

/**
 * 基础请求封装 —— 将 wx.request 包装为 Promise
 * 类比 Java 中的 CompletableFuture 或 RxJava 的 Single
 */
function request(options) {
  return new Promise((resolve, reject) => {
    wx.request({
      url: options.url,
      method: options.method || 'GET',
      data: options.data || {},
      header: options.header || {},
      timeout: options.timeout || 30000,
      dataType: options.dataType || 'json',
      responseType: options.responseType || 'text',
      success(res) {
        // HTTP 状态码检查
        if (res.statusCode >= 200 && res.statusCode < 300) {
          // 业务状态码检查（假设后端统一返回 { code, message, data }）
          if (res.data.code === 0 || res.data.code === 200) {
            resolve(res.data)
          } else {
            // 业务异常（如参数校验失败、权限不足等）
            const error = new Error(res.data.message || '业务异常')
            error.code = res.data.code
            error.data = res.data
            reject(error)
          }
        } else if (res.statusCode === 401) {
          // Token 过期，跳转登录
          handleUnauthorized()
          reject(new Error('登录已过期，请重新登录'))
        } else {
          reject(new Error(`请求失败：HTTP ${res.statusCode}`))
        }
      },
      fail(err) {
        reject(new Error(err.errMsg || '网络请求失败'))
      }
    })
  })
}

/**
 * 处理未授权（401）—— 类比 Spring Security 的 AuthenticationEntryPoint
 */
function handleUnauthorized() {
  // 清除本地 token
  wx.removeStorageSync('token')
  wx.removeStorageSync('userInfo')

  // 跳转到登录页
  const pages = getCurrentPages()
  if (pages[pages.length - 1].route !== 'pages/login/login') {
    wx.redirectTo({ url: '/pages/login/login' })
  }
}

export default request
```

### 11.3 完整的 Request 工具封装（类似 Axios）

以下是一个生产级的请求封装，包含拦截器、Token 管理、错误处理等完整功能。

```javascript
// utils/http.js

// ==================== 配置 ====================
const BASE_CONFIG = {
  baseURL: 'https://api.example.com',
  timeout: 30000,
  contentType: 'application/json'
}

// ==================== 请求拦截器 ====================
const requestInterceptors = []
const responseInterceptors = []

/**
 * 添加请求拦截器 —— 类比 Spring MVC 的 HandlerInterceptor
 * 在请求发出前统一处理（如添加 Token、签名等）
 */
function addRequestInterceptor(onFulfilled, onRejected) {
  requestInterceptors.push({ onFulfilled, onRejected })
}

/**
 * 添加响应拦截器 —— 类比 Spring MVC 的 ResponseBodyAdvice
 * 在响应返回后统一处理（如统一错误提示、数据转换等）
 */
function addResponseInterceptor(onFulfilled, onRejected) {
  responseInterceptors.push({ onFulfilled, onRejected })
}

// ==================== 默认拦截器 ====================

// 请求拦截器：自动添加 Token
addRequestInterceptor(
  (config) => {
    const token = wx.getStorageSync('token')
    if (token) {
      config.header = config.header || {}
      config.header['Authorization'] = `Bearer ${token}`
    }
    return config
  },
  (error) => Promise.reject(error)
)

// 响应拦截器：统一错误提示
addResponseInterceptor(
  (response) => response,
  (error) => {
    if (error.message) {
      wx.showToast({
        title: error.message,
        icon: 'none',
        duration: 2000
      })
    }
    return Promise.reject(error)
  }
)

// ==================== 核心请求方法 ====================

async function http(config) {
  // 合并基础配置
  const finalConfig = {
    ...BASE_CONFIG,
    ...config,
    url: config.url.startsWith('http')
      ? config.url
      : `${BASE_CONFIG.baseURL}${config.url}`,
    header: {
      'content-type': BASE_CONFIG.contentType,
      ...config.header
    }
  }

  try {
    // 执行请求拦截器链
    let chainConfig = finalConfig
    for (const interceptor of requestInterceptors) {
      if (interceptor.onFulfilled) {
        chainConfig = await interceptor.onFulfilled(chainConfig)
      }
    }

    // 发起请求
    const response = await rawRequest(chainConfig)

    // 执行响应拦截器链
    let chainResponse = response
    for (const interceptor of responseInterceptors) {
      if (interceptor.onFulfilled) {
        chainResponse = await interceptor.onFulfilled(chainResponse)
      }
    }

    return chainResponse
  } catch (error) {
    // 执行响应拦截器的错误处理
    for (const interceptor of responseInterceptors) {
      if (interceptor.onRejected) {
        await interceptor.onRejected(error)
      }
    }
    throw error
  }
}

/**
 * 底层 wx.request 的 Promise 包装
 */
function rawRequest(config) {
  return new Promise((resolve, reject) => {
    wx.request({
      url: config.url,
      method: config.method || 'GET',
      data: config.data,
      header: config.header,
      timeout: config.timeout,
      dataType: config.dataType || 'json',
      success(res) {
        if (res.statusCode >= 200 && res.statusCode < 300) {
          if (res.data.code === 0 || res.data.code === 200) {
            resolve(res.data)
          } else {
            const error = new Error(res.data.message || '请求失败')
            error.code = res.data.code
            reject(error)
          }
        } else if (res.statusCode === 401) {
          handleUnauthorized()
          reject(new Error('登录已过期'))
        } else {
          reject(new Error(`服务器错误 (${res.statusCode})`))
        }
      },
      fail(err) {
        reject(new Error(err.errMsg || '网络连接失败'))
      }
    })
  })
}

// ==================== 快捷方法 ====================

function get(url, data, config) {
  return http({ url, method: 'GET', data, ...config })
}

function post(url, data, config) {
  return http({ url, method: 'POST', data, ...config })
}

function put(url, data, config) {
  return http({ url, method: 'PUT', data, ...config })
}

function del(url, data, config) {
  return http({ url, method: 'DELETE', data, ...config })
}

export default { http, get, post, put, del, addRequestInterceptor, addResponseInterceptor }
```

### 11.4 在页面中使用封装后的请求工具

```javascript
// pages/article/detail.js
import http from '../../utils/http'

Page({
  data: {
    article: null,
    comments: [],
    loading: true
  },

  async onLoad(options) {
    const { id } = options
    try {
      // 并行请求文章详情和评论列表
      const [articleRes, commentsRes] = await Promise.all([
        http.get(`/api/articles/${id}`),
        http.get(`/api/articles/${id}/comments`, { page: 1, size: 20 })
      ])

      this.setData({
        article: articleRes.data,
        comments: commentsRes.data.content,
        loading: false
      })
    } catch (error) {
      this.setData({ loading: false })
      console.error('加载失败：', error)
    }
  },

  async submitComment() {
    const content = this.data.commentText.trim()
    if (!content) return

    try {
      const res = await http.post('/api/comments', {
        articleId: this.data.article.id,
        content
      })
      // 将新评论插入列表顶部
      this.setData({
        comments: [res.data, ...this.data.comments],
        commentText: ''
      })
      wx.showToast({ title: '评论成功', icon: 'success' })
    } catch (error) {
      // 错误已由拦截器统一处理
    }
  }
})
```

### 11.5 与 Java 后端 RESTful API 对接

> **Java开发者注意点**：小程序与 Java 后端对接时，需要注意以下要点：
>
> **1. 统一响应格式**
> 后端应统一返回以下格式，方便前端统一处理：
> ```java
> // Spring Boot 统一响应体
> @Data
> @AllArgsConstructor
> public class Result<T> {
>     private int code;       // 0=成功，其他=业务错误码
>     private String message;
>     private T data;
>
>     public static <T> Result<T> success(T data) {
>         return new Result<>(0, "success", data);
>     }
>
>     public static <T> Result<T> error(int code, String message) {
>         return new Result<>(code, message, null);
>     }
> }
> ```
>
> **2. CORS 问题不存在**
> 小程序的 `wx.request` 不是浏览器环境，不存在跨域（CORS）问题。后端无需配置 `@CrossOrigin` 或 CORS Filter。
>
> **3. Content-Type**
> `wx.request` 默认 `content-type` 为 `application/json`，后端直接用 `@RequestBody` 接收即可。如果需要表单提交，需手动设置为 `application/x-www-form-urlencoded`，后端用 `@RequestParam` 接收。
>
> **4. 域名白名单与 HTTPS**
> 正式环境必须配置 HTTPS 域名白名单。在"微信公众平台 -> 开发 -> 开发管理 -> 服务器域名"中配置：
> - request 合法域名：`https://api.yourdomain.com`
> - uploadFile 合法域名：`https://upload.yourdomain.com`
> - downloadFile 合法域名：`https://cdn.yourdomain.com`
