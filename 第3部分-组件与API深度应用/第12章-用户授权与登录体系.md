## 第12章 用户授权与登录体系

用户登录是大多数小程序的必备功能。微信小程序的登录流程与传统的 Web OAuth2 流程有所不同，它使用 `wx.login` 获取临时凭证 `code`，由后端向微信服务器换取用户唯一标识 `openid`，再生成自定义的登录态（如 JWT Token）。

### 12.1 完整登录流程

```
┌──────────────────────────────────────────────────────────────────────────┐
│                        微信小程序登录流程（7步）                            │
├──────────────────────────────────────────────────────────────────────────┤
│                                                                           │
│  ┌─ 1. 小程序端 ─────────────────────────────────────────────────────┐   │
│  │  调用 wx.login() 获取临时登录凭证 code                              │   │
│  └──────────────────────────────────┬─────────────────────────────────┘   │
│                                     ▼                                     │
│  ┌─ 2. 微信服务器 ─────────────────────────────────────────────────────┐   │
│  │  返回 code 给小程序（5分钟有效期）                                  │   │
│  └──────────────────────────────────┬─────────────────────────────────┘   │
│                                     ▼                                     │
│  ┌─ 3. 小程序端 ─────────────────────────────────────────────────────┐   │
│  │  通过 wx.request() 将 code 发送给 Java 后端                        │   │
│  └──────────────────────────────────┬─────────────────────────────────┘   │
│                                     ▼                                     │
│  ┌─ 4. Java 后端 ───────────────────────────────────────────────────┐   │
│  │  用 code + AppID + AppSecret 请求微信服务器                         │   │
│  └──────────────────────────────────┬─────────────────────────────────┘   │
│                                     ▼                                     │
│  ┌─ 5. 微信服务器 ───────────────────────────────────────────────────┐   │
│  │  返回 openid（用户唯一标识）+ session_key                           │   │
│  └──────────────────────────────────┬─────────────────────────────────┘   │
│                                     ▼                                     │
│  ┌─ 6. Java 后端 ───────────────────────────────────────────────────┐   │
│  │  根据 openid 查找/创建用户，生成 JWT Token                          │   │
│  └──────────────────────────────────┬─────────────────────────────────┘   │
│                                     ▼                                     │
│  ┌─ 7. 小程序端 ───────────────────────────────────────────────────┐   │
│  │  接收 Token 并存储到本地，后续请求携带 Token                        │   │
│  └──────────────────────────────────────────────────────────────────┘   │
│                                                                           │
└──────────────────────────────────────────────────────────────────────────┘
```

### 12.2 小程序端登录实现

```javascript
// services/auth.js
import http from '../utils/http'

/**
 * 微信登录 —— 获取 code 并发送给后端
 * 类比 Web 端的 OAuth2 授权码模式
 */
async function wxLogin() {
  try {
    // 第一步：调用 wx.login 获取临时登录凭证 code
    const loginRes = await new Promise((resolve, reject) => {
      wx.login({
        success: resolve,
        fail: reject
      })
    })

    const { code } = loginRes
    if (!code) {
      throw new Error('获取登录凭证失败')
    }

    // 第二步：将 code 发送给 Java 后端
    const res = await http.post('/api/auth/wx-login', { code })

    // 第三步：存储 token 和用户信息
    const { token, userInfo } = res.data
    wx.setStorageSync('token', token)
    wx.setStorageSync('userInfo', userInfo)

    return userInfo
  } catch (error) {
    console.error('登录失败：', error)
    throw error
  }
}

/**
 * 获取用户信息（已废弃，推荐使用头像昵称填写能力）
 * @deprecated 请使用 button open-type="chooseAvatar" 和 input type="nickname"
 * 
 * 新版方案示例：
 * WXML:
 *   <button open-type="chooseAvatar" bindchooseavatar="onChooseAvatar">
 *     <image src="{{avatarUrl}}" />
 *   </button>
 *   <input type="nickname" bindinput="onNicknameInput" />
 * JS:
 *   onChooseAvatar(e) { this.setData({ avatarUrl: e.detail.avatarUrl }) }
 *   onNicknameInput(e) { this.setData({ nickname: e.detail.value }) }
 */
async function getUserProfile() {
  // ⚠️ 此方法已废弃，仅作了解
  // 微信已于2021年4月28日废弃 wx.getUserProfile
  // 请使用新版头像昵称填写能力
  console.warn('wx.getUserProfile 已废弃，请使用头像昵称填写能力')
  
  try {
    const profileRes = await new Promise((resolve, reject) => {
      wx.getUserProfile({
        desc: '用于完善用户资料',
        success: resolve,
        fail: reject
      })
    })

    const { userInfo } = profileRes
    // 将用户信息同步到后端
    await http.put('/api/user/profile', {
      nickname: userInfo.nickName,
      avatarUrl: userInfo.avatarUrl,
      gender: userInfo.gender
    })

    // 更新本地缓存
    const localUser = wx.getStorageSync('userInfo') || {}
    wx.setStorageSync('userInfo', { ...localUser, ...userInfo })

    return userInfo
  } catch (error) {
    console.error('获取用户信息失败：', error)
    throw error
  }
}

/**
 * 检查登录状态，未登录则自动登录
 * 类比 Spring Security 的 SecurityContextHolder
 */
async function ensureLogin() {
  const token = wx.getStorageSync('token')
  if (token) {
    // 验证 token 是否有效
    try {
      await http.get('/api/auth/check')
      return wx.getStorageSync('userInfo')
    } catch (e) {
      // token 无效，清除并重新登录
      wx.removeStorageSync('token')
      wx.removeStorageSync('userInfo')
    }
  }
  // 未登录，执行登录
  return await wxLogin()
}

/**
 * 退出登录
 */
function logout() {
  wx.removeStorageSync('token')
  wx.removeStorageSync('userInfo')
  wx.reLaunch({ url: '/pages/index/index' })
}

export default { wxLogin, getUserProfile, ensureLogin, logout }
```

### 12.3 登录页面实现

```javascript
// pages/login/login.js
import auth from '../../services/auth'

Page({
  data: {
    canIUseGetUserProfile: true
  },

  async onLoginTap() {
    wx.showLoading({ title: '登录中...' })
    try {
      // 第一步：静默登录（获取 openid，生成 token）
      const userInfo = await auth.wxLogin()

      // 第二步：获取用户头像昵称（需要用户点击授权）
      try {
        const profile = await auth.getUserProfile()
        this.setData({ userInfo: profile })
      } catch (e) {
        // 用户拒绝授权，使用默认信息
        console.log('用户拒绝授权头像昵称')
      }

      wx.showToast({ title: '登录成功', icon: 'success' })

      // 返回上一页或跳转首页
      setTimeout(() => {
        const pages = getCurrentPages()
        if (pages.length > 1) {
          wx.navigateBack()
        } else {
          wx.switchTab({ url: '/pages/index/index' })
        }
      }, 1000)
    } catch (error) {
      wx.hideLoading()
      wx.showToast({ title: '登录失败，请重试', icon: 'none' })
    }
  }
})
```

### 12.4 Java 后端 JWT 鉴权对接

> **Java开发者注意点**：以下是你需要在 Spring Boot 后端实现的对应逻辑。

```java
// ====== 1. 微信登录 Controller ======
@RestController
@RequestMapping("/api/auth")
public class AuthController {

    @Autowired
    private WxAuthService wxAuthService;
    @Autowired
    private JwtTokenProvider jwtTokenProvider;

    @PostMapping("/wx-login")
    public Result<Map<String, Object>> wxLogin(@RequestBody Map<String, String> body) {
        String code = body.get("code");

        // 调用微信接口，用 code 换取 openid 和 session_key
        WxSession session = wxAuthService.code2Session(code);
        String openid = session.getOpenid();

        // 查找或创建用户
        User user = wxAuthService.findOrCreateUser(openid);

        // 生成 JWT Token
        String token = jwtTokenProvider.generateToken(user.getId());

        Map<String, Object> data = new HashMap<>();
        data.put("token", token);
        data.put("userInfo", user);

        return Result.success(data);
    }

    @GetMapping("/check")
    public Result<Void> checkToken(@RequestHeader("Authorization") String authHeader) {
        // JWT Filter 已验证，能到这里说明 token 有效
        return Result.success(null);
    }
}

// ====== 2. JWT 工具类 ======
@Component
public class JwtTokenProvider {

    @Value("${jwt.secret}")
    private String secret;

    @Value("${jwt.expiration:86400000}")
    private long expiration; // 默认24小时

    public String generateToken(Long userId) {
        return Jwts.builder()
            .setSubject(String.valueOf(userId))
            .setIssuedAt(new Date())
            .setExpiration(new Date(System.currentTimeMillis() + expiration))
            .signWith(SignatureAlgorithm.HS512, secret)
            .compact();
    }

    public Long getUserIdFromToken(String token) {
        Claims claims = Jwts.parser()
            .setSigningKey(secret)
            .parseClaimsJws(token)
            .getBody();
        return Long.parseLong(claims.getSubject());
    }

    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(secret).parseClaimsJws(token);
            return true;
        } catch (Exception e) {
            return false;
        }
    }
}

// ====== 3. JWT 过滤器 ======
@Component
public class JwtAuthenticationFilter extends OncePerRequestFilter {

    @Autowired
    private JwtTokenProvider jwtTokenProvider;
    @Autowired
    private UserDetailsService userDetailsService;

    @Override
    protected void doFilterInternal(HttpServletRequest request,
                                    HttpServletResponse response,
                                    FilterChain chain)
            throws ServletException, IOException {

        String header = request.getHeader("Authorization");
        if (header != null && header.startsWith("Bearer ")) {
            String token = header.substring(7);
            if (jwtTokenProvider.validateToken(token)) {
                Long userId = jwtTokenProvider.getUserIdFromToken(token);
                UserDetails userDetails = userDetailsService
                    .loadUserByUsername(String.valueOf(userId));
                UsernamePasswordAuthenticationToken auth =
                    new UsernamePasswordAuthenticationToken(
                        userDetails, null, userDetails.getAuthorities());
                SecurityContextHolder.getContext().setAuthentication(auth);
            }
        }
        chain.doFilter(request, response);
    }
}
```

### 12.5 Token 刷新机制

```javascript
// utils/http.js 中添加 Token 刷新逻辑

let isRefreshing = false
let refreshSubscribers = []

function onTokenRefreshed(newToken) {
  refreshSubscribers.forEach(cb => cb(newToken))
  refreshSubscribers = []
}

function addRefreshSubscriber(cb) {
  refreshSubscribers.push(cb)
}

// 在请求拦截器中添加 Token 刷新
addRequestInterceptor(async (config) => {
  const token = wx.getStorageSync('token')
  if (token) {
    config.header['Authorization'] = `Bearer ${token}`
  }
  return config
})

// 在响应拦截器中处理 401
addResponseInterceptor(
  (response) => response,
  async (error) => {
    const originalConfig = error.config

    if (error.message === '登录已过期' && !originalConfig._retry) {
      if (isRefreshing) {
        // 已经在刷新中，将请求加入队列
        return new Promise((resolve) => {
          addRefreshSubscriber((newToken) => {
            originalConfig.header['Authorization'] = `Bearer ${newToken}`
            resolve(http(originalConfig))
          })
        })
      }

      originalConfig._retry = true
      isRefreshing = true

      try {
        // 使用 refreshToken 获取新的 token
        const refreshToken = wx.getStorageSync('refreshToken')
        const res = await rawRequest({
          url: `${BASE_CONFIG.baseURL}/api/auth/refresh`,
          method: 'POST',
          data: { refreshToken },
          header: { 'content-type': 'application/json' }
        })

        const newToken = res.data.token
        wx.setStorageSync('token', newToken)

        onTokenRefreshed(newToken)
        isRefreshing = false

        // 用新 token 重试原请求
        originalConfig.header['Authorization'] = `Bearer ${newToken}`
        return http(originalConfig)
      } catch (refreshError) {
        // 刷新失败，跳转登录
        handleUnauthorized()
        return Promise.reject(refreshError)
      }
    }

    return Promise.reject(error)
  }
)
```

> **类比理解**：Token 刷新机制类似于 Java Web 中 Session 的延长策略：
> - Session 方式：每次请求自动续期 `session.setMaxInactiveInterval()`
> - JWT 方式：Token 过期后，用 Refresh Token 换取新 Token（类似 OAuth2 的 Refresh Token Grant）
> - 上面的实现类似于 Axios 的 Token 刷新队列模式，确保并发请求在 Token 刷新期间不会重复触发刷新。
